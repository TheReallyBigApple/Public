class Plot():
        # note the axes is passed as a reference so no need to return anything
    def setup_axes(ax, productStr=""):
        yearsLoc = mpl.dates.YearLocator()
        monthsLoc = mpl.dates.MonthLocator(interval=2)     # every 2 months
        ax.xaxis.set_major_locator(yearsLoc)    # big ticks for labels
        ax.xaxis.set_minor_locator(monthsLoc) 
        xFmt = mpl.dates.DateFormatter('%Y')
        if (productStr !=""):
            ax.set_title(productStr + ': Past 4 Years',fontsize=15, fontweight='bold', color='blue',pad=40)
        ax.xaxis.set_major_formatter(xFmt)   # format labels on big ticks
        ax.set(ylabel="Rate (%)", xlabel='')
        ax.set_ylim(-1,4)


    def side_by_side(frb1,frb2):
        print("\n\n")
        fig, ax = plt.subplots(1, 2, figsize=(15, 9))
        supTitleStr=frb1.get_productStr() + "/" + frb2.get_productStr() + " past 4 years"
        plt.suptitle(supTitleStr, fontsize=15, fontweight='bold', color='blue', y=1.05)  # y is for padding
        ax[0].plot_date(frb1.data_frame.date, frb1.data_frame.rate, linestyle='-', marker=' ')
        ax[1].plot_date(frb2.data_frame.date, frb2.data_frame.rate, linestyle='-', marker=' ')
        ax[1].get_yaxis().set_visible(False)
        Plot.setup_axes(ax[0], frb1.get_productStr())
        Plot.setup_axes(ax[1], frb2.get_productStr())        
        plt.show()
        print("\n\n")

    def two_together(frb1,frb2):
        print("\n\n")
        fig = plt.figure(figsize=(11,8))
        supTitleStr=frb1.get_productStr() + "/" + frb2.get_productStr() + " past 4 years"
        plt.suptitle(supTitleStr, fontsize=15, fontweight='bold', color='blue', y=1.05)  # y is for padding

        ax.plot_date(frb1.data_frame.date, frb1.data_frame.rate, linestyle='-', marker=' ', label=frb1.get_productStr())

class NY_FRB():
    def __init__(self,product=Product.SOFR, startd="2018-01-01", endd="2022-12-31"):
        self.product=product
        self.startd=startd
        self.endd=endd
        
        self.data_frame=pd.DataFrame()
        

    def get_productStr(self): 
        return Constant.ProductStr[self.product]
    

            
    def create_url(self):
        date_range=self.startd+"&endDt="+self.endd+"&"
        codes="eventCodes="+str(self.product)+"&"+"productCode="+Constant.product_code
        # the 1 in postDt:1 is ascending, -1 is descending
        post_url="&sort=postDt:1,eventCode:1&format="+Constant.format
        url=Constant.base_url+date_range+codes+post_url
        return url

    
    def get_data(self):
        url=self.create_url()
        self.data_frame=pd.read_csv(url)
        self.save_data()
        self.data_frame = self.data_frame.rename(columns={'Effective Date': 'date'})
        self.data_frame = self.data_frame.rename(columns={'Rate (%)': 'rate'})
        self.data_frame = self.data_frame.rename(columns={ 'Volume ($Billions)': 'volume'})
        self.data_frame = self.data_frame[["date", "rate","volume"]]
        # self.data_frame=self.data_frame[::-1].reset_index(drop=True)   # reverse the order
        self.data_frame.date=pd.to_datetime(self.data_frame['date'], format = '%m/%d/%Y')

    def query_data(self, start_date, end_date):
        mask = (self.data_frame['date'] > start_date) & (self.data_frame['date'] <= end_date)
        print(self.get_productStr(), "\n\n")
        display(HTML(self.data_frame.loc[mask].to_html(index=False)))
        

    def save_data(self):
        file_name=Constant.path+Constant.ProductStr[self.product]+".csv"
        self.data_frame.to_csv(file_name)

frb_effr.get_data()
Plot.side_by_side(frb_obfr,frb_effr)

As you can see, both rates trend almost identical. In fact they are almost identical.

EFFR is the median of all Federal Funds transactions, recorded on the previous day.

OBFR is the same but also incldues certain Eurodollar transactions, and certain domestic deposit transactions.

The data for both is from what depository institutions report in the FR 2420 Report of Selected Money Market Rates

Since the median is taken and rounded to the nearest basis point, the rates will often be the same, and often will not change day to day.

For more information, refer to the FRB of NY web site for EFFR and OBFR

SOFR Term Rates




The CME Group administers 1 Month, 3 Month and 6 Month term SOFR rates, that are meant to be more reliable and less volatile than the daily rate.



The CME does not provide free access to the history of these rates.

Tri-Party General Collateral Rate (TGCR) is a measure of rates on overnight, specific-counterparty tri-party general collateral repurchase agreement (repo) transactions secured by Treasury securities.

Broad General Collateral Rate (BGCR) is essentially TGCR plus GCF repo transactions.

Just as we saw wutg OBFR and EFFR, BGCR and TGCR are sourced from the same data with minor differences, making the rates seemingly interchangable.

For more information, refer to the FRB of NY web site for TGCR and BGCR
